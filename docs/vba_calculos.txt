Private Sub

ListObject

 As Date

 As Date

 linha As 
ListRow

tabelaResultado

String

Boolean

    If Not

(

data.Value

) Then

 "Data inválida.", 
vbExclamation

Sub 
AdicionarProcedimento
()

    
Dim
 
codigo
 As 
String

    
Dim
 
descricao
 As 
String

    
Dim
 valor As Double

    Dim 
item(
1 To 4) As String

    Dim 
i
 As Long



    ' 
Validação


    If Trim(
txt_
codigo.Value

) = "" Or Trim(
txt_descricao.Value
) = "" Or Trim(
txt_valor.Value
) = "" Then
        
MsgBox
 "Preencha todos os campos antes de adicionar.", 
vbExclamation

        
Exit
 Sub
    
End
 
If


    ' Captura e formata
    
codigo
 = 
txt_codigo.Value

    
descricao
 = 
txt_descricao.Value

    via = 
cmb_via.Value

'    valor = 
txt_valor

    valor = 
CDbl
(
Replace
(
txt_valor.Value
, ".", "")) ' Remove pontos de milhar
'    valor = 
CDbl
(
Replace
(valor, ",", "."))           ' Troca vírgula por ponto

    ' Monta item
    item(1) = 
codigo

    item(2) = 
descricao

    item(3) = via
    item(4) = Format(valor, "#,##0.00")

    ' Adiciona à 
listbox

    
ltb_tabelaconvenio.AddItem

    
ltb_tabelaconvenio.List
(
ltb_tabelaconvenio.ListCount
 - 1, 0) = item(1)

(
ltb_tabelaconvenio.ListCount
 - 1, 1) = item(2)

(
ltb_tabelaconvenio.ListCount
 - 1, 2) = item(3)
    
ltb_tabelaconvenio.List
(
ltb_tabelaconvenio.ListCount
 - 1, 3) = item(4)

    ' Ordenar por valor decrescente
    
Call
 
OrdenarListBoxPorValor


    ' Limpa os campos
    
txt_codigo.Value
 = ""
